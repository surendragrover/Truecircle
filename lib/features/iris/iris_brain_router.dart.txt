enum IrisBrain { elite, empathy, safety }

class IrisBrainRouter {
  static IrisBrain pickBrain(String text) {
    final t = text.toLowerCase();

    // Safety triggers (basic demo)
    if (t.contains("suicide") ||
        t.contains("kill myself") ||
        t.contains("end my life") ||
        t.contains("self harm") ||
        t.contains("self-harm") ||
        t.contains("मरना") ||
        t.contains("खुदकुशी") ||
        t.contains("जान दे दूँ")) {
      return IrisBrain.safety;
    }

    // Empathy triggers
    if (t.contains("sad") ||
        t.contains("cry") ||
        t.contains("alone") ||
        t.contains("depressed") ||
        t.contains("hopeless") ||
        t.contains("दुख") ||
        t.contains("रो") ||
        t.contains("अकेला") ||
        t.contains("टूट गया")) {
      return IrisBrain.empathy;
    }

    // Elite / CBT triggers
    if (t.contains("anxiety") ||
        t.contains("stress") ||
        t.contains("panic") ||
        t.contains("overthinking") ||
        t.contains("fear") ||
        t.contains("टेंशन") ||
        t.contains("चिंता") ||
        t.contains("घबराहट") ||
        t.contains("ओवरथिंक")) {
      return IrisBrain.elite;
    }

    // Default
    return IrisBrain.empathy;
  }

  static String reply(String userText) {
    final brain = pickBrain(userText);

    switch (brain) {
      case IrisBrain.safety:
        return _safetyReply();

      case IrisBrain.elite:
        return _eliteReply(userText);

      case IrisBrain.empathy:
      default:
        return _empathyReply(userText);
    }
  }

  static String _safetyReply() {
    return "I’m really sorry you’re feeling this way.\n\n"
        "You matter. I can’t help with self-harm, but I can help you stay safe right now.\n\n"
        "If you feel in immediate danger, please contact local emergency services or someone you trust nearby.\n"
        "Right now, do this with me:\n"
        "1) Take 1 slow breath in\n"
        "2) Put both feet on the floor\n"
        "3) Name 5 things you can see\n\n"
        "Tell me: are you safe right now (yes/no)?";
  }

  static String _empathyReply(String userText) {
    return "Thank you for sharing that.\n"
        "What you’re feeling makes sense.\n\n"
        "Let’s keep it gentle and simple:\n"
        "• What happened today that made it feel heavier?\n"
        "• What would feel like even 5% relief right now?";
  }

  static String _eliteReply(String userText) {
    return "Okay. Let’s handle this step-by-step.\n\n"
        "CBT Quick Reset (2 minutes):\n"
        "1) What is the situation? (1 line)\n"
        "2) What thought hit you first?\n"
        "3) What feeling came (0-100%)?\n"
        "4) What is a more balanced thought?\n\n"
        "Write just Step 1 in one sentence.";
  }
}
